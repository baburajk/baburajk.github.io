Direct LVM Mode/Swarm Deployment Testing

Attach disk, configure swarm, deploy using docker-compose the sample flask app. 


# Use Direct LVM mode for Docker 
virsh vol-create-as virspd03 rhce5_docker.img 10G
virsh attach-disk rhce5 --source /data/sdd03/rhce5_docker.img --persistent --target vdc --live

#Enable additional repos
subscription-manager clean
subscription-manager register
subscription-manager attach

subscription-manager repos --list
subscription-manager repos --enable "rhel-*-optional-rpms" --enable "rhel-*-extras-rpms"
yum install https://dl.fedoraproject.org/pub/epel/epel-release-latest-7.noarch.rpm

#Install docker.
yum install docker -y

#Ref : https://docs.docker.com/storage/storagedriver/device-mapper-driver/#configure-direct-lvm-mode-for-production

#Create VG/LV
pvcreate /dev/vdc
vgcreate docker /dev/vdc

lvcreate --wipesignatures y -n thinpool docker -l 95%VG
lvcreate --wipesignatures y -n thinpoolmeta docker -l 1%VG

lvchange --metadataprofile docker-thinpool docker/thinpool
  lvconvert -y \
--zero n \
-c 512K \
--thinpool docker/thinpool \
--poolmetadata docker/thinpoolmeta

[root@rhce5 ~]# cat > /etc/docker/daemon.json < {
>     "storage-driver": "devicemapper",
>     "storage-opts": [
>     "dm.thinpooldev=/dev/mapper/docker-thinpool",
>     "dm.use_deferred_removal=true",
>     "dm.use_deferred_deletion=true"
>     ]
> }
> EOF


[root@rhce5 ~]# cat > /etc/lvm/profile/docker-thinpool.profile <
> activation {
>   thin_pool_autoextend_threshold=80
>   thin_pool_autoextend_percent=20
> }
> EOF


lvchange --metadataprofile docker-thinpool docker/thinpool
lvs -o+seg_monitor
